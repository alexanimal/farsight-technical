src.temporal.pipelines.sector_analysis

Sector analysis pipeline.

This pipeline performs a standardized analysis of funding trends within a sector.

Pipeline steps:

  1. Search for organizations in the sector (semantic search)
  2. Aggregate funding trends over time
  3. Calculate funding velocity metrics
  4. Identify funding patterns
  5. Generate final analysis using sector_trends agent

 23class SectorAnalysisPipeline(PipelineBase):
 24    """Pipeline for analyzing funding trends within a sector.

 27    - Organization discovery via semantic search
 28    - Funding trend aggregation
 29    - Velocity calculation
 30    - Pattern identification
 31    - Final analysis generation
 32    """

 34    def __init__(self, config: Optional[PipelineConfig] = None):
 35        """Initialize the sector analysis pipeline."""
 36        if config is None:
 37            config = PipelineConfig(
 38                name="sector_analysis",
 39                description="Analyze funding trends and patterns within a sector",
 40                version="1.0.0",
 41                metadata={
 42                    "category": "financial_analysis",
 43                    "output_type": "sector_trends_report",
 44                },
 45            )
 46        super().__init__(config)

 48    def build_steps(
 49        self,
 50        context: Dict[str, Any],
 51        config: Optional[Dict[str, Any]] = None,
 52    ) -> List[PipelineStep]:
 53        """Build the steps for sector analysis pipeline.
 54
 55        Args:
 56            context: Pipeline context with:
 57                - sector_name: Name of the sector to analyze
 58                - time_period_start: Start date (ISO format)
 59                - time_period_end: End date (ISO format)
 60                - granularity: Time granularity (monthly/quarterly/yearly)
 61                - min_funding_amount: Optional minimum funding filter
 62            config: Optional configuration overrides.
 63
 64        Returns:
 65            List of PipelineStep objects in execution order.
 66        """
 67        # Extract parameters from context
 68        sector_name = context.get("sector_name", "")
 69        time_period_start = context.get("time_period_start", "")
 70        time_period_end = context.get("time_period_end", "")
 71        granularity = context.get("granularity", "quarterly")
 72        min_funding_amount = context.get("min_funding_amount")
 73
 74        # Merge config overrides if provided
 75        if config:
 76            granularity = config.get("granularity", granularity)
 77            min_funding_amount = config.get("min_funding_amount", min_funding_amount)
 78
 79        steps = []
 80
 81        # Step 1: Semantic search for organizations in sector
 82        steps.append(
 83            PipelineStep(
 84                step_id="step_1_search_orgs",
 85                step_type="tool",
 86                name="semantic_search_organizations",
 87                parameters={
 88                    "text": sector_name,
 89                    "total_funding_usd_min": min_funding_amount,
 90                    "top_k": 100,
 91                },
 92            )
 93        )
 94
 95        # Step 2: Aggregate funding trends
 96        steps.append(
 97            PipelineStep(
 98                step_id="step_2_aggregate_trends",
 99                step_type="tool",
100                name="aggregate_funding_trends",
101                parameters={
102                    # org_uuids will be populated from step_1 result
103                    "time_period_start": time_period_start,
104                    "time_period_end": time_period_end,
105                    "granularity": granularity,
106                    "min_funding_amount": min_funding_amount,
107                },
108            )
109        )
110
111        # Step 3: Calculate funding velocity
112        steps.append(
113            PipelineStep(
114                step_id="step_3_calculate_velocity",
115                step_type="tool",
116                name="calculate_funding_velocity",
117                parameters={
118                    # trend_data will be populated from step_2 result
119                },
120            )
121        )
122
123        # Step 4: Identify funding patterns
124        steps.append(
125            PipelineStep(
126                step_id="step_4_identify_patterns",
127                step_type="tool",
128                name="identify_funding_patterns",
129                parameters={
130                    # trend_data will be populated from step_2 result
131                    "granularity": granularity,
132                    "anomaly_threshold": 2.0,
133                    "detect_seasonality": True,
134                },
135            )
136        )
137
138        # Step 5: Generate final analysis using sector_trends agent
139        steps.append(
140            PipelineStep(
141                step_id="step_5_generate_analysis",
142                step_type="agent",
143                name="sector_trends",
144                parameters={
145                    # Agent will receive accumulated pipeline_data
146                    "query": f"Analyze funding trends for {sector_name}",
147                },
148            )
149        )
150
151        return steps

153    def resolve_step_parameters(
154        self,
155        step: PipelineStep,
156        pipeline_data: Dict[str, Any],
157        step_results: Dict[str, Dict[str, Any]],
158    ) -> Dict[str, Any]:
159        """Resolve step parameters with sector-specific logic.
160
161        Args:
162            step: The PipelineStep to resolve parameters for.
163            pipeline_data: Accumulated data from all previous steps.
164            step_results: Dictionary mapping step_id to step results.
165
166        Returns:
167            Resolved parameters dictionary.
168        """
169        # Call base implementation first
170        resolved = super().resolve_step_parameters(step, pipeline_data, step_results)
171
172        # Sector-specific parameter resolution
173        # Step 2 needs org_uuids from step 1
174        if step.step_id == "step_2_aggregate_trends":
175            if "org_uuids" not in resolved:
176                # Try to get organizations from step_1 result
177                step_1_result = step_results.get("step_1_search_orgs")
178                # semantic_search_organizations returns a list directly, not a dict
179                if isinstance(step_1_result, list):
180                    organizations = step_1_result
181                elif isinstance(step_1_result, dict):
182                    organizations = step_1_result.get("organizations", [])
183                else:
184                    organizations = []
185
186                if organizations:
187                    resolved["org_uuids"] = [
188                        org.get("org_uuid") if isinstance(org, dict) else str(org)
189                        for org in organizations
190                        if org is not None
191                    ]
192                    logger.debug(
193                        f"Resolved {len(resolved['org_uuids'])} org_uuids from step_1 result"
194                    )
195
196        # Step 3 and 4 need trend_data from step 2
197        if step.step_id in ("step_3_calculate_velocity", "step_4_identify_patterns"):
198            if "trend_data" not in resolved:
199                step_2_result = step_results.get("step_2_aggregate_trends", {})
200                if not step_2_result:
201                    # Log warning if step 2 result is missing
202                    logger.warning(
203                        f"Step {step.step_id} requires trend_data from step_2_aggregate_trends, "
204                        f"but step_2 result not found in step_results. "
205                        f"Available step_ids: {list(step_results.keys())}"
206                    )
207                trend_data = step_2_result.get("trend_data", [])
208                if trend_data:
209                    resolved["trend_data"] = trend_data
210                else:
211                    logger.warning(
212                        f"Step {step.step_id} requires trend_data, but step_2 result "
213                        f"does not contain 'trend_data' field. Step_2 result keys: {list(step_2_result.keys())}"
214                    )
215
216        return resolved

221        pipeline_data: Dict[str, Any],
222        context: Dict[str, Any],
223    ) -> Dict[str, Any]:
224        """Build the final consolidated result for sector analysis.
225
226        Args:
227            step_results: Dictionary mapping step_id to step results.
228            pipeline_data: Accumulated data from all steps.
229            context: Original pipeline context.
230
231        Returns:
232            Final consolidated result dictionary.
233        """
234        # Get organizations count from step 1
235        step_1_result = step_results.get("step_1_search_orgs")
236        # semantic_search_organizations returns a list directly, not a dict
237        if isinstance(step_1_result, list):
238            organizations = step_1_result
239        elif isinstance(step_1_result, dict):
240            organizations = step_1_result.get("organizations", [])
241        else:
242            organizations = []
243        organizations_count = len(organizations) if isinstance(organizations, list) else 0
244
245        return {
246            "pipeline_type": "sector_analysis",
247            "pipeline_name": self.name,
248            "pipeline_version": self.config.version,
249            "sector_name": context.get("sector_name"),
250            "time_period": {
251                "start": context.get("time_period_start"),
252                "end": context.get("time_period_end"),
253            },
254            "organizations_found": organizations_count,
255            "trend_analysis": step_results.get("step_2_aggregate_trends", {}),
256            "velocity_metrics": step_results.get("step_3_calculate_velocity", {}),
257            "patterns": step_results.get("step_4_identify_patterns", {}),
258            "final_analysis": step_results.get("step_5_generate_analysis", {}),
259            "metadata": {
260                "steps_executed": len(step_results),
261                "granularity": context.get("granularity", "quarterly"),
262            },
263        }
