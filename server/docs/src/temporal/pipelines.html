src.temporal.pipelines

Pipeline definitions for deterministic analysis pipelines.

This module provides the base pipeline interface and registry for discovering and executing pipeline definitions. Pipelines are separated from workflows to maintain clean separation of concerns.

Pipelines define:

  • Fixed sequences of steps (tools/agents)
  • Step parameter resolution
  • Result consolidation logic

Workflows handle:

  • Temporal execution semantics
  • State management
  • Query/signal handling

 55class PipelineBase(ABC):
 56    """Base class for all pipeline definitions.
 57
 58    Pipelines define:
 59    - A fixed sequence of steps (tools/agents)
 60    - How to resolve step parameters from context and previous results
 61    - How to consolidate final results from step outputs
 62
 63    Pipelines are deterministic and do not contain Temporal-specific logic.
 64    They can be tested in isolation without Temporal.
 65    """

111    def resolve_step_parameters(
112        self,
113        step: PipelineStep,
114        pipeline_data: Dict[str, Any],
115        step_results: Dict[str, Dict[str, Any]],
116    ) -> Dict[str, Any]:
117        """Resolve step parameters that reference previous step results.
118
119        This method allows steps to reference data from previous steps.
120        Override this method to customize parameter resolution logic
121        for specific pipeline types.
122
123        Args:
124            step: The PipelineStep to resolve parameters for.
125            pipeline_data: Accumulated data from all previous steps.
126            step_results: Dictionary mapping step_id to step results.
127
128        Returns:
129            Resolved parameters dictionary.
130        """
131        resolved = {}
132        for key, value in step.parameters.items():
133            if isinstance(value, str) and value.startswith("${") and value.endswith("}"):
134                # Reference to previous step result: ${step_id.field}
135                ref_path = value[2:-1]  # Remove ${ and }
136                parts = ref_path.split(".", 1)
137                if len(parts) == 2:
138                    step_id, field = parts
139                    if step_id in step_results:
140                        step_result = step_results[step_id]
141                        if field in step_result:
142                            resolved[key] = step_result[field]
143                        else:
144                            resolved[key] = value  # Keep unresolved
145                    else:
146                        resolved[key] = value  # Keep unresolved
147                else:
148                    # Direct reference to pipeline_data
149                    if ref_path in pipeline_data:
150                        resolved[key] = pipeline_data[ref_path]
151                    else:
152                        resolved[key] = value  # Keep unresolved
153            else:
154                resolved[key] = value
155
156        return resolved

158    def build_final_result(
159        self,
160        step_results: Dict[str, Dict[str, Any]],
161        pipeline_data: Dict[str, Any],
162        context: Dict[str, Any],
163    ) -> Dict[str, Any]:
164        """Build the final consolidated result from pipeline step results.
165
166        This method consolidates all step results into a final output.
167        Override this method to customize result consolidation for specific
168        pipeline types.
169
170        Args:
171            step_results: Dictionary mapping step_id to step results.
172            pipeline_data: Accumulated data from all steps.
173            context: Original pipeline context.
174
175        Returns:
176            Final consolidated result dictionary.
177        """
178        return {
179            "pipeline_name": self.name,
180            "pipeline_version": self.config.version,
181            "step_results": step_results,
182            "pipeline_data": pipeline_data,
183            "context": context,
184        }

def discover_pipelines() -> None: 
 99def discover_pipelines() -> None:
100    """Discover and register pipelines from the pipelines directory.
101
102    This function attempts to auto-discover pipelines by looking in the
103    src/temporal/pipelines directory. Pipelines should be in:
104    src/temporal/pipelines/{pipeline_name}.py
105
106    This function is defensive and will skip pipelines that can't be imported
107    or don't have the expected structure.
108
109    This function can be called multiple times safely (idempotent).
110    """
111    # Base path: src/temporal/pipelines
112    pipelines_base = Path(__file__).parent
113
114    # Known pipeline names (can be extended)
115    # Pipelines can be registered manually via register_pipeline() if auto-discovery fails
116    known_pipelines = [
117        "sector_analysis",
118        # Add more pipeline names here as they are created
119    ]
120
121    for pipeline_name in known_pipelines:
122        # Skip if already registered
123        if pipeline_name in _PIPELINE_REGISTRY:
124            logger.debug(f"Pipeline {pipeline_name} already registered, skipping")
125            continue
126
127        try:
128            # Try to import the pipeline module
129            module_path = f"src.temporal.pipelines.{pipeline_name}"
130            try:
131                pipeline_module = importlib.import_module(module_path)
132            except (ImportError, ModuleNotFoundError) as e:
133                logger.debug(
134                    f"Could not import pipeline module {module_path}: {e}. "
135                    "Pipeline may not be implemented yet."
136                )
137                continue
138
139            # Try to find the pipeline class
140            # Look for classes that end with "Pipeline" or match the pipeline name
141            pipeline_class = None
142            for attr_name in dir(pipeline_module):
143                if attr_name.startswith("_"):
144                    continue
145                attr = getattr(pipeline_module, attr_name)
146                if (
147                    isinstance(attr, type)
148                    and issubclass(attr, PipelineBase)
149                    and attr != PipelineBase
150                ):
151                    pipeline_class = attr
152                    break
153
154            if pipeline_class is None:
155                logger.debug(
156                    f"No PipelineBase subclass found in {module_path}. "
157                    "Pipeline may not be implemented yet."
158                )
159                continue
160
161            # Register the pipeline
162            register_pipeline(pipeline_name, pipeline_class)
163
164        except Exception as e:
165            logger.warning(f"Failed to register pipeline {pipeline_name} during discovery: {e}")
