 108    @workflow.run
 109    async def run(
 110        self,
 111        context: Dict[str, Any],
 112        agent_plan: Optional[List[str]] = None,
 113        execution_mode: str = "sequential",  # sequential, parallel
 114    ) -> Dict[str, Any]:
 115        """Run the orchestration workflow.
 116
 117        Args:
 118            context: Initial AgentContext as dictionary (query, conversation_id, etc.).
 119            agent_plan: Optional list of agent names to execute in order.
 120                If None, starts with orchestration agent to determine plan.
 121            execution_mode: Execution mode - "sequential" or "parallel".
 122
 123        Returns:
 124            Dictionary containing:
 125            - success: bool indicating if workflow completed successfully
 126            - final_response: Final AgentResponse (if successful)
 127            - agent_responses: List of all agent responses
 128            - workflow_id: Workflow ID
 129            - metadata: Additional metadata
 130        """
 131        # Handle case where Temporal UI passes arguments as a list
 132        # When started from UI, the entire argument array may be passed as first parameter
 133        if isinstance(context, list) and len(context) >= 1:
 134            # Unpack: [context_dict, agent_plan, execution_mode]
 135            context, agent_plan, execution_mode = (
 136                context[0] if len(context) > 0 else {},
 137                context[1] if len(context) > 1 else None,
 138                context[2] if len(context) > 2 else "sequential",
 139            )
 140
 141        # Validate context is a dict
 142        if not isinstance(context, dict):
 143            raise ValueError(
 144                f"Expected context to be a dict, got {type(context).__name__}. "
 145                "Context must be a dictionary with at least a 'query' field."
 146            )
 147
 148        workflow_id = workflow.info().workflow_id
 149        self._state.context = context
 150        self._state.status = WorkflowStatus.RUNNING
 151        self._state.started_at = workflow.now()
 152        self._state.metadata["workflow_id"] = workflow_id
 153        self._state.metadata["execution_mode"] = execution_mode
 154
 155        # Initialize conversation history from context
 156        conversation_history = context.get("conversation_history", [])
 157        if conversation_history:
 158            workflow.logger.info(
 159                f"Loaded {len(conversation_history)} messages from conversation history"
 160            )
 161        else:
 162            conversation_history = []
 163            context["conversation_history"] = conversation_history
 164
 165        # Append user query to conversation history
 166        user_query = context.get("query", "")
 167        if user_query:
 168            conversation_history.append(
 169                {
 170                    "role": "user",
 171                    "content": user_query,
 172                    "timestamp": workflow.now().isoformat() + "Z",
 173                }
 174            )
 175            context["conversation_history"] = conversation_history
 176
 177        workflow.logger.info(
 178            f"Starting orchestrator workflow {workflow_id} with query: {user_query[:100]}"
 179        )
 180
 181        try:
 182            # Iterative reasoning loop
 183            iteration = 0
 184            satisfactory = False
 185            final_response = None
 186
 187            while not satisfactory and iteration < MAX_ITERATIONS:
 188                self._state.current_iteration = iteration
 189                workflow.logger.info(f"Starting iteration {iteration + 1}/{MAX_ITERATIONS}")
 190
 191                # Determine agent plan (initial plan or re-plan)
 192                if iteration == 0:
 193                    # First iteration: use provided plan or determine new plan
 194                    if agent_plan is None:
 195                        # Enrich query before orchestration
 196                        enriched = await self._enrich_query_with_context(context)
 197                        context.update(enriched)
 198                        agent_plan, plan_execution_mode = await self._determine_agent_plan(context)
 199                        # Use execution_mode from orchestration agent's plan if available
 200                        if plan_execution_mode:
 201                            execution_mode = plan_execution_mode
 202                            self._state.metadata["execution_mode"] = execution_mode
 203                            workflow.logger.info(
 204                                f"Using execution_mode from orchestration plan: {execution_mode}"
 205                            )
 206                else:
 207                    # Subsequent iterations: evaluate and decide if replanning is needed
 208                    # Preserve original execution_mode when replanning
 209                    original_execution_mode = execution_mode
 210                    evaluation, replan = await self._evaluate_and_replan(context)
 211
 212                    if evaluation:
 213                        self._state.evaluation_results.append(evaluation)
 214                        satisfactory = evaluation.get("satisfactory", False)
 215                        confidence = evaluation.get("confidence", 0.0)
 216
 217                        workflow.logger.info(
 218                            f"Previous iteration evaluation: "
 219                            f"satisfactory={satisfactory}, confidence={confidence}"
 220                        )
 221
 222                        # Early exit if confidence is very high or answer is satisfactory
 223                        if confidence >= EARLY_EXIT_CONFIDENCE:
 224                            workflow.logger.info(
 225                                f"High confidence ({confidence}) reached, exiting early"
 226                            )
 227                            satisfactory = True
 228                            break
 229                        elif satisfactory:
 230                            workflow.logger.info("Answer is satisfactory, exiting loop")
 231                            break
 232
 233                    # Only replan if not satisfactory and replan is available
 234                    if not satisfactory and replan:
 235                        agent_plan = replan.get("agents", [])
 236                        # Use execution_mode from replan, but fallback to original if not provided
 237                        replan_execution_mode = replan.get("execution_mode")
 238                        if replan_execution_mode:
 239                            execution_mode = replan_execution_mode
 240                        else:
 241                            # Preserve original execution_mode if replan doesn't specify
 242                            execution_mode = original_execution_mode
 243                            workflow.logger.info(
 244                                f"Replan did not specify execution_mode, preserving original: {execution_mode}"
 245                            )
 246                        self._state.metadata["execution_mode"] = execution_mode
 247                        workflow.logger.info(
 248                            f"Re-planning for iteration {iteration + 1}: {len(agent_plan)} agents, "
 249                            f"execution_mode: {execution_mode}"
 250                        )
 251                    elif not satisfactory:
 252                        # Evaluation says not satisfactory but no replan available, exit
 253                        workflow.logger.warning(
 254                            "Answer not satisfactory but no replan available, exiting loop"
 255                        )
 256                        break
 257                    else:
 258                        # Satisfactory, no need to replan
 259                        break
 260
 261                # If no agents to execute, exit loop
 262                if not agent_plan:
 263                    workflow.logger.info("No agents to execute, exiting loop")
 264                    break
 265
 266                # Execute agents according to plan
 267                if execution_mode == "parallel":
 268                    results = await self._execute_agents_parallel(agent_plan, context)
 269                else:
 270                    results = await self._execute_agents_sequential(agent_plan, context)
 271
 272                # Check for cancellation
 273                if self._state.cancellation_requested:
 274                    workflow.logger.info("Cancellation requested, exiting loop")
 275                    break
 276
 277                # Update context with results from this iteration
 278                context = await self._update_context_with_results(context, results, iteration)
 279
 280                # Evaluate response quality after execution (skip on last iteration)
 281                if iteration < MAX_ITERATIONS - 1:
 282                    evaluation = await self._evaluate_response_quality(context, results)
 283                    self._state.evaluation_results.append(evaluation)
 284
 285                    satisfactory = evaluation.get("satisfactory", False)
 286                    confidence = evaluation.get("confidence", 0.0)
 287
 288                    workflow.logger.info(
 289                        f"Iteration {iteration + 1} evaluation: "
 290                        f"satisfactory={satisfactory}, confidence={confidence}"
 291                    )
 292
 293                    # Early exit if confidence is very high
 294                    if confidence >= EARLY_EXIT_CONFIDENCE:
 295                        workflow.logger.info(
 296                            f"High confidence ({confidence}) reached, exiting early"
 297                        )
 298                        satisfactory = True
 299                else:
 300                    # Last iteration: don't evaluate, just mark as satisfactory
 301                    workflow.logger.info("Last iteration, skipping evaluation")
 302                    satisfactory = True
 303
 304                iteration += 1
 305
 306            # After loop: consolidate final response
 307            workflow.logger.info(
 308                f"Completed {iteration} iteration(s), consolidating final response"
 309            )
 310
 311            # Consolidate agent responses into final answer
 312            if self._state.agent_responses:
 313                # Filter out orchestration agent responses from planning/evaluation phases
 314                specialized_agent_responses = [
 315                    resp
 316                    for resp in self._state.agent_responses
 317                    if resp.get("agent_category") != "orchestration"
 318                ]
 319
 320                # If we have specialized agent responses, consolidate them
 321                if specialized_agent_responses:
 322                    try:
 323                        workflow.logger.info(
 324                            f"Consolidating {len(specialized_agent_responses)} agent response(s)"
 325                        )
 326
 327                        # Create consolidation context
 328                        consolidation_context = {
 329                            "query": context.get("query", ""),
 330                            "conversation_history": context.get("conversation_history", []),
 331                            "metadata": {
 332                                "mode": "consolidation",
 333                                "agent_responses": specialized_agent_responses,
 334                            },
 335                        }
 336
 337                        # Call orchestration agent in consolidation mode
 338                        consolidation_result = await self._execute_single_agent(
 339                            "orchestration", consolidation_context
 340                        )
 341
 342                        if consolidation_result.get("success"):
 343                            consolidated_response = consolidation_result.get("response", {})
 344                            final_response = consolidated_response
 345                            workflow.logger.info("Successfully consolidated agent responses")
 346
 347                            # Append consolidated response to conversation history
 348                            await self._append_to_history("assistant", consolidated_response)
 349                        else:
 350                            workflow.logger.warning(
 351                                f"Consolidation failed: {consolidation_result.get('error', 'Unknown error')}"
 352                            )
 353                            # Fallback to last agent response
 354                            final_response = self._state.agent_responses[-1]
 355                            # Append fallback response to history
 356                            if final_response:
 357                                await self._append_to_history("assistant", final_response)
 358                    except Exception as e:
 359                        workflow.logger.error(f"Error during consolidation: {e}", exc_info=True)
 360                        # Fallback to last agent response
 361                        final_response = self._state.agent_responses[-1]
 362                        # Append fallback response to history
 363                        if final_response:
 364                            await self._append_to_history("assistant", final_response)
 365                else:
 366                    # No specialized agent responses, use orchestration response if available
 367                    for response in reversed(self._state.agent_responses):
 368                        if response.get("agent_category") == "orchestration":
 369                            final_response = response
 370                            break
 371                    if final_response is None and self._state.agent_responses:
 372                        final_response = self._state.agent_responses[-1]
 373                    # Append final response to history
 374                    if final_response:
 375                        await self._append_to_history("assistant", final_response)
 376
 377            # Determine final status
 378            if self._state.cancellation_requested:
 379                self._state.status = WorkflowStatus.CANCELLED
 380                self._state.completed_at = workflow.now()
 381                self._state.metadata["iterations_completed"] = iteration
 382                return {
 383                    "success": False,
 384                    "final_response": final_response,
 385                    "agent_responses": self._state.agent_responses,
 386                    "workflow_id": workflow_id,
 387                    "metadata": {
 388                        "cancelled": True,
 389                        "cancellation_reason": self._state.cancellation_reason,
 390                        "iterations_completed": iteration,
 391                    },
 392                }
 393
 394            # Check if any agents failed
 395            failed_agents = [
 396                name
 397                for name, exec_state in self._state.agent_executions.items()
 398                if exec_state.status == "failed"
 399            ]
 400
 401            if failed_agents:
 402                self._state.status = WorkflowStatus.FAILED
 403                self._state.error = f"Agents failed: {', '.join(failed_agents)}"
 404            else:
 405                self._state.status = WorkflowStatus.COMPLETED
 406
 407            self._state.completed_at = workflow.now()
 408            self._state.metadata["iterations_completed"] = iteration
 409            self._state.metadata["final_satisfactory"] = satisfactory
 410
 411            return {
 412                "success": self._state.status == WorkflowStatus.COMPLETED,
 413                "final_response": final_response,
 414                "agent_responses": self._state.agent_responses,
 415                "workflow_id": workflow_id,
 416                "metadata": self._state.metadata,
 417            }
 418
 419        except Exception as e:
 420            error_msg = f"Workflow execution failed: {str(e)}"
 421            workflow.logger.error(error_msg, exc_info=True)
 422            self._state.status = WorkflowStatus.FAILED
 423            self._state.error = error_msg
 424            self._state.completed_at = workflow.now()
 425            raise
 426
 427        finally:
 428            # Save conversation history after workflow completion
 429            await self._save_conversation_history()

 431    async def _determine_agent_plan(
 432        self, context: Dict[str, Any]
 433    ) -> Tuple[List[str], Optional[str]]:
 434        """Determine which agents to execute using orchestration agent.
 435
 436        Args:
 437            context: Initial context.
 438
 439        Returns:
 440            Tuple of (agent_plan, execution_mode):
 441            - agent_plan: List of agent names to execute
 442            - execution_mode: Execution mode from orchestration plan, or None if not found
 443        """
 444        workflow.logger.info("Determining agent plan using orchestration agent")
 445
 446        # Execute orchestration agent to determine plan
 447        result = await self._execute_single_agent("orchestration", context)
 448
 449        if not result.get("success"):
 450            # Fallback: if orchestration fails, return empty plan
 451            workflow.logger.warning("Orchestration agent failed, using empty plan")
 452            return [], None
 453
 454        # Extract agent plan and execution_mode from orchestration response metadata
 455        response = result.get("response", {})
 456        metadata = response.get("metadata", {})
 457        execution_plan = metadata.get("execution_plan", {})
 458
 459        if isinstance(execution_plan, dict) and "agents" in execution_plan:
 460            agent_plan = execution_plan["agents"]
 461            # Extract execution_mode from the plan
 462            execution_mode = execution_plan.get("execution_mode")
 463            workflow.logger.info(
 464                f"Determined agent plan: {agent_plan}, execution_mode: {execution_mode}"
 465            )
 466            return agent_plan, execution_mode
 467        else:
 468            workflow.logger.warning("No execution plan found in orchestration response metadata")
 469            return [], None

 471    async def _evaluate_and_replan(
 472        self, context: Dict[str, Any]
 473    ) -> Tuple[Optional[Dict[str, Any]], Optional[Dict[str, Any]]]:
 474        """Evaluate current responses and create a re-plan if needed.
 475
 476        Args:
 477            context: Current context with agent responses.
 478
 479        Returns:
 480            Tuple of (evaluation, replan):
 481            - evaluation: Evaluation result dict with satisfactory, confidence, etc., or None if failed
 482            - replan: Dictionary with "agents" and "execution_mode" if re-planning needed,
 483              None if no re-plan needed or evaluation failed
 484        """
 485        # Get all agent responses from all iterations
 486        all_responses = self._state.agent_responses.copy()
 487
 488        # Filter out orchestration agent responses (planning/evaluation)
 489        specialized_responses = [
 490            resp for resp in all_responses if resp.get("agent_category") != "orchestration"
 491        ]
 492
 493        if not specialized_responses:
 494            workflow.logger.warning("No specialized agent responses for evaluation")
 495            return None, None
 496
 497        # Ensure query is enriched (if not already)
 498        if "extracted_entities" not in context.get("metadata", {}):
 499            enriched = await self._enrich_query_with_context(context)
 500            context.update(enriched)
 501
 502        # Create evaluation context
 503        evaluation_context = {
 504            "query": context.get("query", ""),
 505            "conversation_history": context.get("conversation_history", []),
 506            "metadata": {
 507                "mode": "evaluation_and_replanning",
 508                "agent_responses": specialized_responses,
 509                "execution_history": self._state.execution_history,
 510            },
 511        }
 512
 513        # Call orchestration agent in evaluation mode
 514        result = await self._execute_single_agent("orchestration", evaluation_context)
 515
 516        if not result.get("success"):
 517            workflow.logger.warning("Evaluation failed, cannot create re-plan")
 518            return None, None
 519
 520        response = result.get("response", {})
 521        metadata = response.get("metadata", {})
 522        evaluation = metadata.get("evaluation")
 523        replan = metadata.get("replan")
 524
 525        if evaluation:
 526            workflow.logger.info(
 527                f"Evaluation result: satisfactory={evaluation.get('satisfactory')}, "
 528                f"confidence={evaluation.get('confidence')}"
 529            )
 530        else:
 531            workflow.logger.warning("No evaluation result found in orchestration response")
 532
 533        if replan and isinstance(replan, dict) and "agents" in replan:
 534            workflow.logger.info(
 535                f"Re-plan created: {len(replan['agents'])} agents, "
 536                f"mode: {replan.get('execution_mode', 'sequential')}"
 537            )
 538            return evaluation, replan
 539        else:
 540            workflow.logger.info("Evaluation determined no re-planning needed")
 541            return evaluation, None

 610    async def _update_context_with_results(
 611        self, context: Dict[str, Any], results: List[Dict[str, Any]], iteration: int
 612    ) -> Dict[str, Any]:
 613        """Update context with results from an iteration.
 614
 615        This method accumulates agent responses, updates shared data,
 616        and prepares context for the next iteration.
 617
 618        Args:
 619            context: Current context.
 620            results: Results from agent execution.
 621            iteration: Current iteration number.
 622
 623        Returns:
 624            Updated context dictionary.
 625        """
 626        # Store iteration responses in metadata
 627        iteration_key = f"iteration_{iteration}_responses"
 628        if "metadata" not in context:
 629            context["metadata"] = {}
 630        context["metadata"][iteration_key] = results
 631
 632        # Accumulate shared data from all successful results
 633        for result in results:
 634            if result.get("success") and result.get("response"):
 635                response = result["response"]
 636                # Update shared data
 637                if "shared_data" in response.get("metadata", {}):
 638                    context.setdefault("shared_data", {}).update(
 639                        response["metadata"]["shared_data"]
 640                    )
 641
 642        # Add iteration info to metadata
 643        context["metadata"]["current_iteration"] = iteration
 644        context["metadata"]["total_iterations"] = MAX_ITERATIONS
 645
 646        return context

 648    async def _execute_single_agent(
 649        self, agent_name: str, context: Dict[str, Any]
 650    ) -> Dict[str, Any]:
 651        """Execute a single agent and update state.
 652
 653        Args:
 654            agent_name: Name of agent to execute.
 655            context: Context to pass to agent.
 656
 657        Returns:
 658            Execution result dictionary.
 659        """
 660        # Create execution state
 661        exec_state = AgentExecutionState(
 662            agent_name=agent_name,
 663            agent_category="unknown",  # Will be updated after execution
 664            status="running",
 665            started_at=workflow.now(),
 666        )
 667        self._state.agent_executions[agent_name] = exec_state
 668
 669        workflow.logger.info(f"Executing agent: {agent_name}")
 670
 671        try:
 672            # Execute agent via activity
 673            # Use activity name as string to avoid importing activities module
 674            # (which would pull in non-deterministic dependencies like langfuse/httpx)
 675            result = await workflow.execute_activity(
 676                "execute_agent",
 677                args=[agent_name, context],
 678                start_to_close_timeout=timedelta(seconds=600),
 679                retry_policy=RetryPolicy(
 680                    initial_interval=timedelta(seconds=2),
 681                    backoff_coefficient=2.0,
 682                    maximum_interval=timedelta(seconds=120),
 683                    maximum_attempts=3,
 684                ),
 685            )
 686
 687            exec_state.completed_at = workflow.now()
 688
 689            if result.get("success"):
 690                exec_state.status = "completed"
 691                response = result.get("response", {})
 692                exec_state.response = response
 693                exec_state.agent_category = response.get("agent_category", "unknown")
 694
 695                # Add response to state
 696                self._state.agent_responses.append(response)
 697
 698                # Update shared data if agent provides it
 699                if "shared_data" in response.get("metadata", {}):
 700                    self._state.shared_data.update(response["metadata"]["shared_data"])
 701
 702                # Add to execution history
 703                self._state.execution_history.append(
 704                    {
 705                        "agent_name": agent_name,
 706                        "status": "completed",
 707                        "timestamp": exec_state.completed_at.isoformat(),
 708                    }
 709                )
 710            else:
 711                exec_state.status = "failed"
 712                exec_state.error = result.get("error", "Unknown error")
 713
 714                # Add to execution history
 715                self._state.execution_history.append(
 716                    {
 717                        "agent_name": agent_name,
 718                        "status": "failed",
 719                        "error": exec_state.error,
 720                        "timestamp": exec_state.completed_at.isoformat(),
 721                    }
 722                )
 723
 724            return result
 725
 726        except Exception as e:
 727            error_msg = f"Agent {agent_name} execution raised exception: {str(e)}"
 728            workflow.logger.error(error_msg, exc_info=True)
 729            exec_state.status = "failed"
 730            exec_state.error = error_msg
 731            exec_state.completed_at = workflow.now()
 732
 733            self._state.execution_history.append(
 734                {
 735                    "agent_name": agent_name,
 736                    "status": "failed",
 737                    "error": error_msg,
 738                    "timestamp": exec_state.completed_at.isoformat(),
 739                }
 740            )
 741
 742            return {
 743                "success": False,
 744                "error": error_msg,
 745                "agent_name": agent_name,
 746                "response": None,
 747            }

 749    async def _execute_agents_sequential(
 750        self, agent_plan: List[str], initial_context: Dict[str, Any]
 751    ) -> List[Dict[str, Any]]:
 752        """Execute agents sequentially, passing context between them.
 753
 754        Args:
 755            agent_plan: List of agent names to execute.
 756            initial_context: Initial context.
 757
 758        Returns:
 759            List of execution results.

163                pipeline_type = input_dict.get("pipeline_type", "")
