 63        self.config = self._load_config()

 68        Returns:
 69            AgentConfig: Parsed configuration object

 71        Raises:
 72            ValueError: If the config file is invalid or missing required fields
 73        """
 74        try:
 75            with open(self.config_path, "r", encoding="utf-8") as f:

104        """Add a message to the conversation history.

125    def get_conversation_history_for_llm(
126        self, max_messages: Optional[int] = 50
127    ) -> List[Dict[str, str]]:
128        """Get conversation history formatted for LLM prompts.
129
130        This method returns the conversation history in a format suitable for
131        including in LLM prompts. It optionally truncates to the most
132        recent messages to prevent token limit issues.
133
134        Args:
135            max_messages: Maximum number of messages to return. If None,
136                returns all messages. Defaults to 50.
137
138        Returns:
139            List of message dictionaries with 'role' and 'content' keys.
140            Returns empty list if no history exists.
141        """
142        if self.conversation_history is None or len(self.conversation_history) == 0:
143            return []
144
145        # If max_messages is None, return all
146        if max_messages is None:
147            return self.conversation_history.copy()
148
149        # Return last N messages (most recent)
150        return self.conversation_history[-max_messages:].copy()

 53    This class represents the output from an agent, including the response

 56
 57    Attributes:

 60        agent_name: The name of the agent that generated this response.
 61        agent_category: The category of the agent that generated this response.
 62        tool_calls: Optional list of tool calls that were made during processing.
 63            Each tool call is a dict with 'name', 'parameters', and optionally 'result'.

 66        nested_responses: Optional list of AgentResponse objects from coordinated agents.
