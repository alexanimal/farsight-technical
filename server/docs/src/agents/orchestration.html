 339    async def execute(self, context: AgentContext) -> AgentOutput:
 340        """Execute the orchestration agent in planning or consolidation mode.
 341
 342        This method supports three modes:
 343        1. **Planning mode** (default): Creates an execution plan
 344        2. **Consolidation mode**: Consolidates agent insights into final answer
 345        3. **Evaluation mode**: Evaluates response quality and creates re-plan if needed
 346
 347        Args:
 348            context: The agent context containing the user query and metadata.
 349                - For planning: metadata may contain "available_agents"
 350                - For consolidation: metadata must contain "agent_responses" (List[AgentOutput])
 351
 352        Returns:
 353            AgentOutput containing:
 354            - Planning mode: Dict with "agents" (list of agent names) and "execution_mode"
 355            - Consolidation mode: AgentInsight with consolidated final answer
 356            - status: SUCCESS if operation completed successfully
 357            - metadata: Additional information about the process
 358        """
 359        try:
 360            # Check mode from metadata
 361            mode = context.get_metadata("mode")
 362
 363            if mode == "consolidation":
 364                # Consolidation mode: combine agent insights
 365                logger.info("Orchestration agent in consolidation mode")
 366                agent_responses = context.get_metadata("agent_responses", [])
 367
 368                if not agent_responses:
 369                    logger.warning("No agent responses provided for consolidation")
 370                    return create_agent_output(
 371                        content="",
 372                        agent_name=self.name,
 373                        agent_category=self.category,
 374                        status=ResponseStatus.ERROR,
 375                        error="No agent responses provided for consolidation",
 376                    )
 377
 378                # Consolidate insights
 379                consolidated_insight = await self._consolidate_responses(context, agent_responses)
 380
 381                logger.info("Orchestration agent completed consolidation")
 382
 383                return create_agent_output(
 384                    content=consolidated_insight,
 385                    agent_name=self.name,
 386                    agent_category=self.category,
 387                    status=ResponseStatus.SUCCESS,
 388                    metadata={
 389                        "query": context.query,
 390                        "mode": "consolidation",
 391                        "num_agents_consolidated": len(agent_responses),
 392                    },
 393                )
 394            elif mode == "evaluation_and_replanning":
 395                # Evaluation mode: evaluate response quality and create re-plan
 396                logger.info("Orchestration agent in evaluation and replanning mode")
 397                agent_responses = context.get_metadata("agent_responses", [])
 398                execution_history = context.get_metadata("execution_history", [])
 399
 400                if not agent_responses:
 401                    logger.warning("No agent responses provided for evaluation")
 402                    return create_agent_output(
 403                        content="",
 404                        agent_name=self.name,
 405                        agent_category=self.category,
 406                        status=ResponseStatus.ERROR,
 407                        error="No agent responses provided for evaluation",
 408                    )
 409
 410                # Evaluate response quality
 411                evaluation = await self._evaluate_response_quality(
 412                    context, agent_responses, execution_history
 413                )
 414
 415                # If not satisfactory, create re-plan
 416                if not evaluation.get("satisfactory", False):
 417                    logger.info("Response not satisfactory, creating re-plan")
 418                    replan = await self._create_replan(context, evaluation, agent_responses)
 419
 420                    return create_agent_output(
 421                        content="",
 422                        agent_name=self.name,
 423                        agent_category=self.category,
 424                        status=ResponseStatus.SUCCESS,
 425                        metadata={
 426                            "query": context.query,
 427                            "mode": "evaluation_and_replanning",
 428                            "evaluation": evaluation,
 429                            "replan": replan,
 430                        },
 431                    )
 432                else:
 433                    # Response is satisfactory
 434                    logger.info("Response is satisfactory, no re-planning needed")
 435                    return create_agent_output(
 436                        content="",
 437                        agent_name=self.name,
 438                        agent_category=self.category,
 439                        status=ResponseStatus.SUCCESS,
 440                        metadata={
 441                            "query": context.query,
 442                            "mode": "evaluation_and_replanning",
 443                            "evaluation": evaluation,
 444                            "replan": None,
 445                        },
 446                    )
 447            else:
 448                # Planning mode (default): create execution plan
 449                logger.info(f"Orchestration agent analyzing query: {context.query[:100]}")
 450
 451                # Get available agents from metadata, or discover dynamically, or use fallback

 454                )

 718    def _validate_and_normalize_plan(
 719        self, plan_data: Dict[str, Any], available_agents: Dict[str, Dict[str, Any]]
 720    ) -> Dict[str, Any]:
 721        """Validate and normalize the execution plan from LLM.
 722
 723        Args:
 724            plan_data: Raw plan data from LLM.
 725            available_agents: Dictionary of available agents.
 726
 727        Returns:
 728            Validated and normalized plan dictionary.
 729        """
 730        # Extract agents list
 731        agents = plan_data.get("agents", [])
 732        if not isinstance(agents, list):
 733            agents = []
 734
 735        # Filter to only include valid agent names
 736        valid_agents = [agent for agent in agents if agent in available_agents]
 737
 738        # Get reasoning
 739        reasoning = plan_data.get("reasoning", "")
 740        if not isinstance(reasoning, str):
 741            reasoning = ""
 742
 743        # Safety check: If reasoning mentions an agent but it's not in the agents list, add it
 744        # This handles cases where the LLM recognizes relevance but doesn't include the agent
 745        if reasoning and not valid_agents:
 746            reasoning_lower = reasoning.lower()
 747            for agent_name in available_agents.keys():
 748                # Check if agent name or its description keywords are mentioned in reasoning
 749                agent_info = available_agents[agent_name]
 750                agent_desc_lower = agent_info.get("description", "").lower()
 751
 752                # Check if agent name appears in reasoning
 753                if agent_name.lower() in reasoning_lower:
 754                    if agent_name not in valid_agents:
 755                        logger.info(
 756                            f"Found agent '{agent_name}' mentioned in reasoning but not in agents list. Adding it."
 757                        )
 758                        valid_agents.append(agent_name)
 759                # Also check if key terms from the agent's description appear
 760                elif any(
 761                    keyword.lower() in reasoning_lower for keyword in agent_info.get("keywords", [])
 762                ):
 763                    if agent_name not in valid_agents:
 764                        logger.info(
 765                            f"Found agent '{agent_name}' keywords in reasoning but not in agents list. Adding it."
 766                        )
 767                        valid_agents.append(agent_name)
 768
 769        # Validate execution mode
 770        execution_mode = plan_data.get("execution_mode", "sequential")
 771        if execution_mode not in ["sequential", "parallel"]:
 772            execution_mode = "sequential"
 773
 774        # Validate confidence
 775        confidence = plan_data.get("confidence", 0.5)
 776        if not isinstance(confidence, (int, float)) or confidence < 0.0 or confidence > 1.0:
 777            confidence = 0.5
 778
 779        return {
 780            "agents": valid_agents,
 781            "execution_mode": execution_mode,
 782            "reasoning": reasoning,
 783            "confidence": float(confidence),
 784        }

 786    def _create_fallback_plan(
 787        self, context: AgentContext, available_agents: Dict[str, Dict[str, Any]]
 788    ) -> Dict[str, Any]:
 789        """Create a fallback plan using keyword matching.
 790
 791        This is used when LLM planning fails.
 792
 793        Args:
 794            context: The agent context.
 795            available_agents: Dictionary of available agents.
 796
 797        Returns:
 798            Fallback plan dictionary.
 799        """
 800        query_lower = context.query.lower()
 801        selected_agents = []
 802
 803        # Simple keyword matching
 804        for agent_name, agent_info in available_agents.items():
 805            keywords = agent_info.get("keywords", [])
 806            if any(keyword.lower() in query_lower for keyword in keywords):
 807                selected_agents.append(agent_name)
 808
 809        # Default to sequential execution
 810        execution_mode = "sequential"
 811
 812        # If multiple agents selected, consider parallel if they're independent
 813        if len(selected_agents) > 1:
 814            # For now, default to sequential for safety
 815            # This could be enhanced with more sophisticated logic
 816            execution_mode = "sequential"
 817
 818        return {
 819            "agents": selected_agents,
 820            "execution_mode": execution_mode,
 821            "reasoning": f"Fallback plan based on keyword matching in query",
 822            "confidence": 0.6,
 823        }
