343    async def _extract_parameters(self, context: AgentContext) -> Dict[str, Any]:
344        """Extract parameters from user query using LLM function calling.
345
346        Args:
347            context: The agent context containing the user query.
348
349        Returns:
350            Dictionary of extracted parameters.
351        """
352        # Use pre-extracted metadata when available
353        extracted = context.get_metadata("extracted_entities", {})
354        params: Dict[str, Any] = {}
355
356        # Handle investors structure: can be dict with "names" key or list (defensive coding)
357        investors_data = extracted.get("investors", {})
358        if isinstance(investors_data, dict):
359            investor_names = investors_data.get("names", [])
360        elif isinstance(investors_data, list):
361            investor_names = investors_data
362        else:
363            investor_names = []
364
365        if investor_names:
366            params["investor_name"] = investor_names[0]
367
368        time_period = extracted.get("time_period", {})
369        if isinstance(time_period, dict):
370            if time_period.get("start"):
371                params["time_period_start"] = time_period["start"]
372            if time_period.get("end"):
373                params["time_period_end"] = time_period["end"]
374
375        if params and params.get("investor_name"):
376            # Set defaults for time period if not provided
377            if not params.get("time_period_start"):
378                five_years_ago = datetime.now() - timedelta(days=1825)
379                params["time_period_start"] = five_years_ago.strftime("%Y-%m-%dT00:00:00")
380            if not params.get("time_period_end"):
381                params["time_period_end"] = datetime.now().strftime("%Y-%m-%dT23:59:59")
382            return params
383
384        extract_params_tool = {
385            "type": "function",
386            "function": {
387                "name": "extract_investor_analysis_params",
388                "description": "Extract investor name, time period, and other parameters from a query about investor portfolio performance.",
389                "parameters": {
390                    "type": "object",
391                    "properties": {
392                        "investor_name": {
393                            "type": "string",
394                            "description": "Name of the investor to analyze (e.g., 'Sequoia Capital', 'Andreessen Horowitz', 'Accel Partners'). Required.",
395                        },
396                        "time_period_start": {
397                            "type": "string",
398                            "description": "Start date for analysis period in ISO format (e.g., '2018-01-01T00:00:00'). Defaults to 5 years ago if not specified.",
399                        },
400                        "time_period_end": {
401                            "type": "string",
402                            "description": "End date for analysis period in ISO format (e.g., '2024-12-31T23:59:59'). Defaults to today if not specified.",
403                        },
404                        "include_exits_only": {
405                            "type": "boolean",
406                            "description": "Whether to only analyze exited investments. Default: false.",
407                        },
408                    },
409                    "required": ["investor_name"],
410                },
411            },
412        }
413
414        # Build system prompt using prompt manager
415        prompt_manager = get_prompt_manager()
416        system_prompt = prompt_manager.build_system_prompt(
417            base_prompt=self._extract_params_prompt,
418            options=PromptOptions(add_temporal_context=True, add_markdown_instructions=True),
419        )
420
421        # Build user prompt using prompt manager
422        user_prompt_content = f"""User Query: {context.query}
423
424Extract the investor name, time period, and other parameters from this query about investor portfolio performance."""
425
426        user_prompt = prompt_manager.build_user_prompt(
427            user_query=user_prompt_content,
428            options=PromptOptions(add_temporal_context=True),
429        )
430
431        try:
432            result = await generate_llm_function_response(
433                prompt=user_prompt,
434                tools=[extract_params_tool],
435                system_prompt=system_prompt,
436                model="gpt-4.1-mini",
437                temperature=0.3,
438                tool_choice={
439                    "type": "function",
440                    "function": {"name": "extract_investor_analysis_params"},
441                },
442            )
443
444            if isinstance(result, dict) and "function_name" in result:
445                if result["function_name"] == "extract_investor_analysis_params":
446                    params = result["arguments"]
447                    # Set defaults for time period if not provided
448                    if not params.get("time_period_start"):
449                        five_years_ago = datetime.now() - timedelta(days=1825)
450                        params["time_period_start"] = five_years_ago.strftime("%Y-%m-%dT00:00:00")
451                    if not params.get("time_period_end"):
452                        params["time_period_end"] = datetime.now().strftime("%Y-%m-%dT23:59:59")
453                    if "include_exits_only" not in params:
454                        params["include_exits_only"] = False
455                    # Filter out None values
456                    return {k: v for k, v in params.items() if v is not None}
457                else:
458                    logger.warning(
459                        f"Unexpected function call: {result.get('function_name')}, using defaults"
460                    )
461            else:
462                logger.warning(f"LLM did not make expected function call. Got: {type(result)}")
463        except Exception as e:
464            logger.warning(f"LLM parameter extraction failed: {e}, using defaults", exc_info=True)
465
466        # Return defaults if extraction failed
467        five_years_ago = datetime.now() - timedelta(days=1825)
468        return {
469            "investor_name": None,  # Will cause error if not found
470            "time_period_start": five_years_ago.strftime("%Y-%m-%dT00:00:00"),
471            "time_period_end": datetime.now().strftime("%Y-%m-%dT23:59:59"),
472            "include_exits_only": False,
473        }

476        self,

536Your task:
537- Summarize overall portfolio performance (exit rate, ROI, time to exit)
538- Analyze sector concentration and diversification
539- Identify investment patterns (preferred stages, velocity, timing)
540- Compare performance to market benchmarks
541- Highlight strengths and weaknesses
542- State uncertainty where data is limited
543- Directly answer the user's query in the summary
544
545Focus on actionable insights for finance professionals: due diligence, benchmarking, strategy insights, relationship building."""
546
547        system_prompt = prompt_manager.build_system_prompt(
548            base_prompt=base_prompt,
549            options=PromptOptions(add_temporal_context=True, add_markdown_instructions=True),
550        )
551
552        # Build user prompt with data
553        user_prompt_content = f"""User Query: {context.query}
554
555Investor: {investor_name}
556
557Portfolio Data (JSON):
558{json.dumps(portfolio_data, indent=2, default=str)}
559
560Portfolio Metrics (JSON):
561{json.dumps(portfolio_metrics, indent=2, default=str)}
562
563Sector Concentration (JSON):
564{json.dumps(sector_concentration, indent=2, default=str)}
565
566Investment Patterns (JSON):
567{json.dumps(investment_patterns, indent=2, default=str)}
568
569Benchmark Comparison (JSON):
570{json.dumps(benchmark_comparison, indent=2, default=str)}
571
572Analyze this data and generate insights that directly answer the user's query about investor portfolio performance. Call the generate_insight function with your analysis."""
573
574        user_prompt = prompt_manager.build_user_prompt(
575            user_query=user_prompt_content,
576            options=PromptOptions(add_temporal_context=True),
577        )
578
579        try:
580            # Call LLM with function calling
581            result = await generate_llm_function_response(
582                prompt=user_prompt,
583                tools=[generate_insight_tool],
584                system_prompt=system_prompt,
585                model="gpt-4.1-mini",
586                temperature=0.3,
587                tool_choice={
588                    "type": "function",
589                    "function": {"name": "generate_insight"},
590                },
591            )
592
593            # Extract and return AgentInsight
594            if isinstance(result, dict) and "function_name" in result:
595                args = result["arguments"]
596                return AgentInsight(
597                    summary=args["summary"],

602            else:

606                    confidence=0.0,
607                )
608        except Exception as e:
609            logger.warning(f"Failed to generate insight from LLM: {e}", exc_info=True)
610            # Fallback insight
611            return AgentInsight(
612                summary=f"Analyzed portfolio for {investor_name} but encountered an error generating insights.",
613                confidence=0.0,

Execute the investor analysis agent to analyze portfolio performance.

This method:

  1. Extracts investor name and time parameters from the query
  2. Uses find_investor_portfolio to discover portfolio companies
  3. Calculates portfolio performance metrics
  4. Analyzes sector concentration
  5. Identifies investment patterns
  6. Compares to market benchmarks
  7. Formats the results into a natural language response

Args: context: The agent context containing the user query and metadata.

Returns: AgentOutput containing: - content: Natural language response with portfolio analysis - status: SUCCESS if query processed successfully - metadata: Information about the analysis and results - tool_calls: List of tool calls made during execution
