src.temporal

Temporal integration layer for durable workflow orchestration.

This module provides the Temporal integration for the system, including:

  • Client for API layer to interact with workflows
  • Worker process for executing workflows and activities
  • Signal and query definitions
  • Workflow and activity implementations

Architecture:

  • API layer should only import from this module's client interface
  • Workflows and activities are registered by the worker process
  • Signals and queries provide type-safe communication

136    async def start_workflow(
137        self,
138        context: Dict[str, Any],
139        agent_plan: Optional[List[str]] = None,
140        execution_mode: str = "sequential",
141        workflow_id: Optional[str] = None,
142        workflow_timeout: Optional[timedelta] = None,
143    ) -> str:
144        """Start an orchestrator workflow.
145
146        Args:
147            context: AgentContext as dictionary (query, conversation_id, etc.).
148            agent_plan: Optional list of agent names to execute. If None,
149                workflow will use orchestration agent to determine plan.
150            execution_mode: Execution mode - "sequential" or "parallel".
151            workflow_id: Optional workflow ID. If not provided, Temporal will
152                generate one.
153            workflow_timeout: Optional workflow execution timeout.
154
155        Returns:
156            The workflow ID.
157
158        Raises:
159            RuntimeError: If client is not connected.
160            Exception: If workflow start fails.
161        """
162        client = self._ensure_connected()
163
164        # Validate context has required fields
165        if "query" not in context:
166            raise ValueError("Context must contain 'query' field")
167
168        # Import workflow lazily to avoid import chain issues during workflow validation
169        from src.temporal.workflows.orchestrator import OrchestratorWorkflow
170
171        try:
172            # Temporal requires an 'id' parameter - generate one if not provided
173            if workflow_id is None:
174                workflow_id = f"workflow-{uuid.uuid4().hex[:12]}"
175
176            # Build kwargs for start_workflow
177            workflow_kwargs: Dict[str, Any] = {
178                "id": workflow_id,
179                "args": [context, agent_plan, execution_mode],
180                "task_queue": self._task_queue,
181            }
182            if workflow_timeout is not None:
183                workflow_kwargs["execution_timeout"] = workflow_timeout
184
185            handle = await client.start_workflow(
186                OrchestratorWorkflow.run,
187                **workflow_kwargs,
188            )
189
190            logger.info(f"Started workflow: {handle.id}")
191            return handle.id
192
193        except Exception as e:
194            logger.error(f"Failed to start workflow: {e}", exc_info=True)
195            raise

221        try:

223            await handle.signal(SIGNAL_CANCELLATION, signal)
224            logger.info(f"Sent cancellation signal to workflow: {workflow_id}")
225        except Exception as e:
226            # Check if it's a workflow not found error
227            error_str = str(e).lower()
228            if "not found" in error_str or "workflow" in error_str:
229                logger.warning(f"Workflow not found: {workflow_id}")
230                # Re-raise as a more specific exception for API layer

232            logger.error(f"Failed to send cancellation signal: {e}", exc_info=True)
