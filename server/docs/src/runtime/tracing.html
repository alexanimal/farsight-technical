src.runtime.tracing

Tracing module for agent operations using Langfuse.

This module provides the system's memory of what happened. Agent systems are non-linear, parallel, and probabilistic — traces are essential for debugging, evaluation, auditing, and replay.

It provides:

  • Span creation
  • Context propagation
  • Correlation IDs
  • Structured metadata

 35def _convert_to_langfuse_trace_id(trace_id: str) -> str:
 36    """Convert a trace ID to Langfuse format (32 lowercase hex chars, no dashes).
 37
 38    Langfuse expects trace IDs to be exactly 32 lowercase hexadecimal characters
 39    without dashes. UUIDs have dashes, so we remove them.
 40
 41    Args:
 42        trace_id: Trace ID in any format (UUID, hex string, etc.)
 43
 44    Returns:
 45        Trace ID in Langfuse format (32 lowercase hex chars, no dashes)
 46    """
 47    # Remove dashes and convert to lowercase
 48    cleaned = trace_id.replace("-", "").lower()
 49
 50    # If it's already 32 chars, return it
 51    if len(cleaned) == 32:
 52        return cleaned
 53
 54    # If it's a UUID (36 chars with dashes removed = 32), return it
 55    # Otherwise, pad or truncate to 32 chars (though this shouldn't happen with UUIDs)
 56    if len(cleaned) > 32:
 57        return cleaned[:32]
 58    elif len(cleaned) < 32:
 59        # Pad with zeros if needed (shouldn't happen with UUIDs)
 60        return cleaned.ljust(32, "0")
 61
 62    return cleaned

136    @contextmanager
137    def span(
138        self,
139        name: str,
140        trace_id: Optional[str] = None,
141        parent_observation_id: Optional[str] = None,
142        metadata: Optional[Dict[str, Any]] = None,
143    ):
144        """Create a tracing span context manager.
145
146        Args:
147            name: Name of the span.
148            trace_id: Optional trace ID. If None, creates a new one.
149            parent_observation_id: Optional parent observation ID for nesting.
150            metadata: Optional metadata to attach to the span.
151
152        Yields:
153            The observation object for adding additional metadata.
154        """
155        if not self.enabled or self.client is None:
156            yield None
157            return
158
159        trace_id = trace_id or self.create_trace_id()
160        # Langfuse v3 API - use start_as_current_observation() which replaces deprecated start_as_current_span()
161        # First, ensure trace context is set if we have a trace_id
162        if trace_id:
163            # Set trace context for Langfuse to use
164            current_context = self.get_trace_context() or {}
165            current_context["trace_id"] = trace_id
166            self.set_trace_context(current_context)
167
168        # Prepare trace_context for Langfuse if we have a trace_id
169        trace_context = None
170        if trace_id:
171            try:
172                from langfuse.types import TraceContext
173
174                # Convert UUID to Langfuse format (32 hex chars, no dashes)
175                langfuse_trace_id = _convert_to_langfuse_trace_id(trace_id)
176                trace_context = TraceContext(trace_id=langfuse_trace_id, parent_span_id=None)
177            except (ImportError, AttributeError):
178                # Fallback: Langfuse will use its internal context management
179                pass
180            except Exception as e:
181                # Log but don't fail if trace context creation fails
182                logger.debug(
183                    f"Could not create TraceContext: {e}, using internal context management"
184                )
185                trace_context = None
186
187        try:
188            # Use start_as_current_observation() which replaces deprecated start_as_current_span()
189            # Parent relationships are handled automatically through Langfuse's current span context
190            # This returns a context manager that automatically handles the observation lifecycle
191            with self.client.start_as_current_observation(
192                name=name,
193                as_type="span",
194                metadata=metadata or {},
195                trace_context=trace_context,
196            ) as observation:
197                # Set trace context for our internal tracking
198                context = {
199                    "trace_id": trace_id,
200                    "span_name": name,
201                    "observation_id": (observation.id if hasattr(observation, "id") else None),
202                }
203                if parent_observation_id:
204                    context["parent_observation_id"] = parent_observation_id
205
206                self.set_trace_context(context)
207
208                yield observation
209        except (AttributeError, TypeError) as e:
210            # Fallback if method doesn't exist or has wrong signature
211            logger.warning(
212                f"Langfuse start_as_current_observation() not available: {e}, tracing disabled for span: {name}"
213            )
214            yield None
215            return
216        except Exception as e:
217            logger.warning(f"Error in span {name}: {e}")
218            yield None
219            return

221    @asynccontextmanager
222    async def async_span(
223        self,
224        name: str,
225        trace_id: Optional[str] = None,
226        parent_observation_id: Optional[str] = None,
227        metadata: Optional[Dict[str, Any]] = None,
228    ):
229        """Create an async tracing span context manager.
230
231        Args:
232            name: Name of the span.
233            trace_id: Optional trace ID. If None, creates a new one.
234            parent_observation_id: Optional parent observation ID for nesting.
235            metadata: Optional metadata to attach to the span.
236
237        Yields:
238            The observation object for adding additional metadata.
239        """
240        if not self.enabled or self.client is None:
241            yield None
242            return
243
244        trace_id = trace_id or self.create_trace_id()
245        # Langfuse v3 API - use start_as_current_observation() which replaces deprecated start_as_current_span()
246        # First, ensure trace context is set if we have a trace_id
247        if trace_id:
248            # Set trace context for Langfuse to use
249            current_context = self.get_trace_context() or {}
250            current_context["trace_id"] = trace_id
251            self.set_trace_context(current_context)
252
253        # Prepare trace_context for Langfuse if we have a trace_id
254        trace_context = None
255        if trace_id:
256            try:
257                from langfuse.types import TraceContext
258
259                # Convert UUID to Langfuse format (32 hex chars, no dashes)
260                langfuse_trace_id = _convert_to_langfuse_trace_id(trace_id)
261                trace_context = TraceContext(trace_id=langfuse_trace_id, parent_span_id=None)
262            except (ImportError, AttributeError):
263                # Fallback: Langfuse will use its internal context management
264                pass
265            except Exception as e:
266                # Log but don't fail if trace context creation fails
267                logger.debug(
268                    f"Could not create TraceContext: {e}, using internal context management"
269                )
270                trace_context = None
271
272        try:
273            # Use start_as_current_observation() which replaces deprecated start_as_current_span()
274            # Parent relationships are handled automatically through Langfuse's current span context
275            # Note: start_as_current_observation() returns a sync context manager, so we need
276            # to wrap it to work in async context by manually calling __enter__ and __exit__
277            sync_cm = self.client.start_as_current_observation(
278                name=name,
279                as_type="span",
280                metadata=metadata or {},
281                trace_context=trace_context,
282            )
283
284            # Enter the sync context manager (this is safe to do in async code as it's lightweight)
285            observation = sync_cm.__enter__()
286
287            exc_type = None
288            exc_val = None
289            exc_tb = None
290            try:
291                # Set trace context for our internal tracking
292                context = {
293                    "trace_id": trace_id,
294                    "span_name": name,
295                    "observation_id": (observation.id if hasattr(observation, "id") else None),
296                }
297                if parent_observation_id:
298                    context["parent_observation_id"] = parent_observation_id
299
300                self.set_trace_context(context)
301
302                yield observation
303            except Exception:
304                # Capture exception info to pass to __exit__
305                exc_type, exc_val, exc_tb = sys.exc_info()
306                raise
307            finally:
308                # Exit the sync context manager, passing exception info if any
309                sync_cm.__exit__(exc_type, exc_val, exc_tb)
310        except (AttributeError, TypeError) as e:
311            # Fallback if method doesn't exist or has wrong signature
312            logger.warning(
313                f"Langfuse start_as_current_observation() not available: {e}, tracing disabled for span: {name}"
314            )
315            yield None
316            return
317        except Exception as e:
318            logger.warning(f"Error in async span {name}: {e}")
319            yield None
320            return

322    def log_event(
323        self,
324        name: str,
325        trace_id: Optional[str] = None,
326        parent_observation_id: Optional[str] = None,
327        metadata: Optional[Dict[str, Any]] = None,
328    ) -> None:
329        """Log an event to the trace.
330
331        Args:
332            name: Name of the event.
333            trace_id: Optional trace ID. If None, uses current context or creates new.
334            parent_observation_id: Optional parent observation ID.
335            metadata: Optional metadata for the event.
336        """
337        if not self.enabled:
338            return
339
340        # Get trace_id from context if not provided
341        if trace_id is None:
342            context = self.get_trace_context()
343            trace_id = context.get("trace_id") if context else None
344
345        trace_id = trace_id or self.create_trace_id()
346
347        if self.client is None:
348            return
349
350        # Set trace context so Langfuse can use it
351        if trace_id:
352            current_context = self.get_trace_context() or {}
353            current_context["trace_id"] = trace_id
354            self.set_trace_context(current_context)
355
356        try:
357            # Langfuse v3 API - use create_event() method
358            # trace_id is managed through Langfuse context, not passed as parameter
359            # Note: parent_observation_id is not a valid parameter for create_event
360            event_metadata = metadata or {}
361            if parent_observation_id:
362                event_metadata["parent_observation_id"] = parent_observation_id
363            self.client.create_event(
364                name=name,
365                metadata=event_metadata,
366            )
367        except (AttributeError, TypeError) as e:
368            logger.warning(f"Langfuse create_event() not available: {e}, event not logged: {name}")
369        except Exception as e:
370            logger.warning(f"Error logging event {name}: {e}")

373        self,
374        name: str,

377        trace_id: Optional[str] = None,
378        parent_observation_id: Optional[str] = None,

380    ) -> None:
381        """Log a generation (e.g., LLM call) to the trace.
382
383        Args:
384            name: Name of the generation.
385            input_data: Input data for the generation.
386            output_data: Output data from the generation.
387            trace_id: Optional trace ID. If None, uses current context or creates new.
388            parent_observation_id: Optional parent observation ID.
389            metadata: Optional metadata for the generation.
390        """
391        if not self.enabled:
392            return
393
394        # Get trace_id from context if not provided
395        if trace_id is None:
396            context = self.get_trace_context()
397            trace_id = context.get("trace_id") if context else None
398
399        trace_id = trace_id or self.create_trace_id()
400
401        if self.client is None:
402            return
403
404        # Set trace context so Langfuse can use it
405        if trace_id:
406            current_context = self.get_trace_context() or {}
407            current_context["trace_id"] = trace_id
408            self.set_trace_context(current_context)
409
410        try:
411            # Langfuse v3 API - use start_generation() method
412            # trace_id is managed through Langfuse context, not passed as parameter
413            # Note: start_generation() returns a generation object that needs to be ended
414            # Note: parent_observation_id is not a valid parameter for start_generation
415            generation_metadata = metadata or {}
416            if parent_observation_id:
417                generation_metadata["parent_observation_id"] = parent_observation_id
418            generation = self.client.start_generation(
419                name=name,
420                metadata=generation_metadata,
421            )
422            # Set input and output on the generation object
423            if hasattr(generation, "update"):
424                generation.update(input=input_data, output=output_data)
425            # End the generation
426            if hasattr(generation, "end"):
427                generation.end()
428        except (AttributeError, TypeError) as e:
429            logger.warning(
430                f"Langfuse start_generation() not available: {e}, generation not logged: {name}"
431            )
432        except Exception as e:
433            logger.warning(f"Error logging generation {name}: {e}")
