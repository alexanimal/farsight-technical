src.runtime.executor

Execution substrate for agent operations.

This module provides the Executor class that serves as the "hands of the system". It executes agent invocations and tool calls, enforces execution contracts (timeouts, cancellation), and wraps execution with tracing.

The executor provides a stable interface so orchestration logic is runtime-agnostic.

 57    ):
 58        """Initialize the executor.
 59
 60        Args:
 61            default_timeout: Default timeout in seconds for executions.

 63            enable_tracing: Whether to enable tracing for executions.

 68    async def execute_agent(
 69        self,
 70        agent: AgentBase,
 71        agent_input: AgentInput,
 72        timeout: Optional[float] = None,
 73        cancellation_token: Optional[asyncio.CancelledError] = None,
 74    ) -> AgentOutput:
 75        """Execute an agent invocation.
 76
 77        Args:
 78            agent: The agent instance to execute. Must have an async `execute` method
 79                that takes AgentContext and returns AgentOutput.
 80            agent_input: The validated agent input contract.
 81            timeout: Timeout in seconds. If None, uses default_timeout.
 82            cancellation_token: Optional cancellation token for cancellation propagation.
 83
 84        Returns:
 85            AgentOutput: The response from the agent execution.
 86
 87        Raises:
 88            TimeoutError: If execution exceeds the timeout.
 89            asyncio.CancelledError: If execution is cancelled.
 90            ValueError: If agent doesn't have an execute method or input is invalid.
 91        """
 92        timeout = timeout or self.default_timeout
 93
 94        # Verify agent has execute method
 95        if not hasattr(agent, "execute") or not callable(getattr(agent, "execute")):
 96            raise ValueError(
 97                f"Agent {agent.name} does not have an async execute method. "
 98                "Agents must implement async def execute(context: AgentContext) -> AgentOutput"
 99            )
100
101        # Validate input using contracts
102        validated_input = validate_agent_input(agent_input.to_agent_context())
103        agent_context = validated_input.to_agent_context()
104
105        # Create execution metadata
106        execution_metadata = {
107            "agent_name": agent.name,
108            "agent_category": agent.category,
109            "timeout": timeout,
110        }
111
112        # Get or create trace context
113        tracer = get_tracer()
114        trace_context = tracer.get_trace_context()
115        trace_id = trace_context.get("trace_id") if trace_context else tracer.create_trace_id()
116        parent_observation_id = trace_context.get("observation_id") if trace_context else None
117
118        try:
119            # Wrap execution with tracing if enabled
120            if self.enable_tracing:
121                execution_metadata["trace_id"] = trace_id
122                execution_metadata["span_started"] = True
123
124                async with tracer.async_span(
125                    name=f"agent.{agent.name}",
126                    trace_id=trace_id,
127                    parent_observation_id=parent_observation_id,
128                    metadata={
129                        "agent_name": agent.name,
130                        "agent_category": agent.category,
131                        "query": (
132                            agent_context.query[:100]
133                            if len(agent_context.query) > 100
134                            else agent_context.query
135                        ),
136                        "timeout": timeout,
137                    },
138                ) as span:
139                    # Set input on the observation
140                    if span is not None and hasattr(span, "update"):
141                        try:
142                            # Convert agent_input to dict for Langfuse
143                            input_dict = (
144                                agent_input.model_dump()
145                                if hasattr(agent_input, "model_dump")
146                                else {
147                                    "query": agent_context.query,
148                                    "conversation_id": agent_context.conversation_id,
149                                    "user_id": agent_context.user_id,
150                                    "metadata": agent_context.metadata,
151                                    "shared_data": agent_context.shared_data,
152                                }
153                            )
154                            span.update(input=input_dict)
155                        except Exception as e:
156                            logger.debug(f"Failed to update span input: {e}")
157
158                    try:
159                        # Execute with timeout if specified
160                        if timeout is not None:
161                            output = await asyncio.wait_for(
162                                self._execute_with_cancellation(
163                                    agent, agent_context, cancellation_token
164                                ),
165                                timeout=timeout,
166                            )
167                        else:
168                            output = await self._execute_with_cancellation(
169                                agent, agent_context, cancellation_token
170                            )
171
172                        # Ensure we have AgentOutput
173                        if not isinstance(output, AgentOutput):
174                            raise ValueError(
175                                f"Agent {agent.name} returned {type(output)}, expected AgentOutput"
176                            )
177
178                        # Set output on the observation
179                        if span is not None and hasattr(span, "update"):
180                            try:
181                                # Convert AgentOutput to dict for Langfuse
182                                output_dict = (
183                                    output.model_dump()
184                                    if hasattr(output, "model_dump")
185                                    else {
186                                        "content": (
187                                            str(output.content)
188                                            if hasattr(output, "content")
189                                            else ""
190                                        ),
191                                        "status": (
192                                            output.status.value
193                                            if hasattr(output.status, "value")
194                                            else str(output.status)
195                                        ),
196                                        "agent_name": output.agent_name,
197                                        "agent_category": output.agent_category,
198                                        "error": (
199                                            output.error if hasattr(output, "error") else None
200                                        ),
201                                    }
202                                )
203                                span.update(output=output_dict)
204                            except Exception as e:
205                                logger.debug(f"Failed to update span output: {e}")
206
207                        # Add execution metadata to output
208                        output.metadata.update(execution_metadata)
209                        execution_metadata["span_completed"] = True
210
211                        return output
212                    except Exception as e:
213                        # Capture errors that occur during execution and update the observation
214                        error_msg = str(e)
215                        if span is not None and hasattr(span, "update"):
216                            try:
217                                span.update(
218                                    output={
219                                        "error": error_msg,
220                                        "error_type": type(e).__name__,
221                                        "status": "error",
222                                    }
223                                )
224                            except Exception as update_error:
225                                logger.debug(f"Failed to update span with error: {update_error}")
226                        # Re-raise to be handled by outer exception handlers
227                        raise
228            else:
229                # Execute without tracing
230                if timeout is not None:
231                    output = await asyncio.wait_for(
232                        self._execute_with_cancellation(agent, agent_context, cancellation_token),
233                        timeout=timeout,
234                    )
235                else:
236                    output = await self._execute_with_cancellation(
237                        agent, agent_context, cancellation_token
238                    )
239
240                # Ensure we have AgentOutput
241                if not isinstance(output, AgentOutput):
242                    raise ValueError(
243                        f"Agent {agent.name} returned {type(output)}, expected AgentOutput"
244                    )
245
246                # Add execution metadata to output
247                output.metadata.update(execution_metadata)
248
249                return output
250
251        except asyncio.TimeoutError as e:
252            error_msg = f"Agent {agent.name} execution timed out after {timeout}s"
253            logger.error(error_msg)
254            if self.enable_tracing:
255                execution_metadata["span_error"] = True
256                execution_metadata["error"] = error_msg
257                tracer.log_event(
258                    name="agent.timeout",
259                    trace_id=trace_id,
260                    parent_observation_id=parent_observation_id,
261                    metadata={
262                        "agent_name": agent.name,
263                        "timeout": timeout,
264                        "error": error_msg,
265                    },
266                )
267
268            return create_agent_output(
269                content="",
270                agent_name=agent.name,
271                agent_category=agent.category,
272                status=ResponseStatus.ERROR,
273                error=error_msg,
274                metadata=execution_metadata,
275            )
276
277        except asyncio.CancelledError as e:
278            error_msg = f"Agent {agent.name} execution was cancelled"
279            logger.warning(error_msg)
280            if self.enable_tracing:
281                execution_metadata["span_cancelled"] = True
282                execution_metadata["error"] = error_msg
283                tracer.log_event(
284                    name="agent.cancelled",
285                    trace_id=trace_id,
286                    parent_observation_id=parent_observation_id,
287                    metadata={
288                        "agent_name": agent.name,
289                        "error": error_msg,
290                    },
291                )
292
293            return create_agent_output(
294                content="",
295                agent_name=agent.name,
296                agent_category=agent.category,
297                status=ResponseStatus.ERROR,
298                error=error_msg,
299                metadata=execution_metadata,
300            )
301
302        except Exception as e:
303            error_msg = f"Agent {agent.name} execution failed: {str(e)}"
304            logger.error(error_msg, exc_info=True)
305            if self.enable_tracing:
306                execution_metadata["span_error"] = True
307                execution_metadata["error"] = error_msg
308                tracer.log_event(
309                    name="agent.error",
310                    trace_id=trace_id,
311                    parent_observation_id=parent_observation_id,
312                    metadata={
313                        "agent_name": agent.name,
314                        "error": error_msg,
315                        "error_type": type(e).__name__,
316                    },
317                )
318
319            return create_agent_output(
320                content="",
321                agent_name=agent.name,
322                agent_category=agent.category,
323                status=ResponseStatus.ERROR,
324                error=error_msg,
325                metadata=execution_metadata,
326            )

328    async def execute_tool(
329        self,
330        tool_input: ToolInput,
331        tool_func: Callable[..., Awaitable[Any]],
332        timeout: Optional[float] = None,
333        cancellation_token: Optional[asyncio.CancelledError] = None,
334        tool_metadata: Optional[ToolMetadata] = None,
335    ) -> ToolOutput:
336        """Execute a tool call.
337
338        Args:
339            tool_input: The validated tool input contract.
340            tool_func: Async callable that implements the tool.
341            timeout: Timeout in seconds. If None, uses default_timeout.
342            cancellation_token: Optional cancellation token for cancellation propagation.
343            tool_metadata: Optional ToolMetadata for parameter validation. If provided,
344                parameters will be validated against the tool's schema.
345
346        Returns:
347            ToolOutput: The structured output from tool execution.
348
349        Raises:
350            TimeoutError: If execution exceeds the timeout.
351            asyncio.CancelledError: If execution is cancelled.
352            ValueError: If parameters don't match the tool's schema (when tool_metadata is provided).
353        """
354        timeout = timeout or self.default_timeout
355
356        # Validate ToolInput
357        if tool_metadata:
358            validated_input = validate_tool_input(
359                tool_input.tool_name,
360                tool_input.parameters,
361                tool_metadata,
362            )
363        else:
364            validated_input = validate_tool_input(
365                tool_input.tool_name,
366                tool_input.parameters,
367            )
368
369        tool_name = validated_input.tool_name
370        tool_params = validated_input.parameters
371        input_metadata = validated_input.metadata
372
373        # Merge input metadata with provided metadata
374        execution_metadata = {
375            "tool_name": tool_name,
376            "parameters": tool_params,
377            "timeout": timeout,
378            **input_metadata,
379        }
380        if tool_metadata:
381            execution_metadata["tool_metadata"] = tool_metadata.model_dump()
382
383        # Get or create trace context
384        tracer = get_tracer()
385        trace_context = tracer.get_trace_context()
386        trace_id = trace_context.get("trace_id") if trace_context else tracer.create_trace_id()
387        parent_observation_id = trace_context.get("observation_id") if trace_context else None
388
389        # Track execution time for ToolOutput
390        start_time = time.time()
391
392        try:
393            if self.enable_tracing:
394                execution_metadata["trace_id"] = trace_id
395                execution_metadata["span_started"] = True
396
397                async with tracer.async_span(
398                    name=f"tool.{tool_name}",
399                    trace_id=trace_id,
400                    parent_observation_id=parent_observation_id,
401                    metadata={
402                        "tool_name": tool_name,
403                        "parameters": tool_params,
404                        "timeout": timeout,
405                    },
406                ) as span:
407                    # Set input on the observation
408                    if span is not None and hasattr(span, "update"):
409                        try:
410                            span.update(
411                                input={
412                                    "tool_name": tool_name,
413                                    "parameters": tool_params,
414                                }
415                            )
416                        except Exception as e:
417                            logger.debug(f"Failed to update span input: {e}")
418
419                    try:
420                        # Execute with timeout if specified
421                        if timeout is not None:
422                            result = await asyncio.wait_for(
423                                self._execute_tool_with_cancellation(
424                                    tool_func, tool_params, cancellation_token
425                                ),
426                                timeout=timeout,
427                            )
428                        else:
429                            result = await self._execute_tool_with_cancellation(
430                                tool_func, tool_params, cancellation_token
431                            )
432
433                        # Set output on the observation
434                        if span is not None and hasattr(span, "update"):
435                            try:
436                                # Convert result to a serializable format
437                                if isinstance(result, dict):
438                                    output_data = result
439                                elif hasattr(result, "model_dump"):
440                                    output_data = result.model_dump()
441                                else:
442                                    output_data = {"result": result}
443                                span.update(output=output_data)
444                            except Exception as e:
445                                logger.debug(f"Failed to update span output: {e}")
446
447                        execution_metadata["span_completed"] = True
448
449                        # Calculate execution time
450                        execution_time_ms = (time.time() - start_time) * 1000
451
452                        # Return ToolOutput
453                        return create_tool_output(
454                            tool_name=tool_name,
455                            success=True,
456                            result=result,
457                            execution_time_ms=execution_time_ms,
458                            metadata=execution_metadata,
459                        )
460                    except Exception as e:
461                        # Capture errors that occur during execution and update the observation
462                        error_msg = str(e)
463                        if span is not None and hasattr(span, "update"):
464                            try:
465                                span.update(
466                                    output={
467                                        "error": error_msg,
468                                        "error_type": type(e).__name__,
469                                        "success": False,
470                                    }
471                                )
472                            except Exception as update_error:
473                                logger.debug(f"Failed to update span with error: {update_error}")
474                        # Re-raise to be handled by outer exception handlers
475                        raise
476            else:
477                # Execute without tracing
478                if timeout is not None:
479                    result = await asyncio.wait_for(
480                        self._execute_tool_with_cancellation(
481                            tool_func, tool_params, cancellation_token
482                        ),
483                        timeout=timeout,
484                    )
485                else:
486                    result = await self._execute_tool_with_cancellation(
487                        tool_func, tool_params, cancellation_token
488                    )
489
490                # Calculate execution time
491                execution_time_ms = (time.time() - start_time) * 1000
492
493                # Return ToolOutput
494                return create_tool_output(
495                    tool_name=tool_name,
496                    success=True,
497                    result=result,
498                    execution_time_ms=execution_time_ms,
499                    metadata=execution_metadata,
500                )
501
502        except asyncio.TimeoutError:
503            error_msg = f"Tool {tool_name} execution timed out after {timeout}s"
504            logger.error(error_msg)
505            execution_time_ms = (time.time() - start_time) * 1000
506
507            if self.enable_tracing:
508                execution_metadata["span_error"] = True
509                tracer.log_event(
510                    name="tool.timeout",
511                    trace_id=trace_id,
512                    parent_observation_id=parent_observation_id,
513                    metadata={
514                        "tool_name": tool_name,
515                        "timeout": timeout,
516                        "error": error_msg,
517                    },
518                )
519
520            # Return ToolOutput
521            return create_tool_output(
522                tool_name=tool_name,
523                success=False,
524                error=error_msg,
525                execution_time_ms=execution_time_ms,
526                metadata=execution_metadata,
527            )
528
529        except asyncio.CancelledError:
530            error_msg = f"Tool {tool_name} execution was cancelled"
531            logger.warning(error_msg)
532            execution_time_ms = (time.time() - start_time) * 1000
533
534            if self.enable_tracing:
535                execution_metadata["span_cancelled"] = True
536                tracer.log_event(
537                    name="tool.cancelled",
538                    trace_id=trace_id,
539                    parent_observation_id=parent_observation_id,
540                    metadata={
541                        "tool_name": tool_name,
542                        "error": error_msg,
543                    },
544                )
545
546            # Return ToolOutput
547            return create_tool_output(
548                tool_name=tool_name,
549                success=False,
550                error=error_msg,
551                execution_time_ms=execution_time_ms,
552                metadata=execution_metadata,
553            )
554
555        except Exception as e:
556            error_msg = f"Tool {tool_name} execution failed: {str(e)}"
557            logger.error(error_msg, exc_info=True)
558            execution_time_ms = (time.time() - start_time) * 1000
559
560            if self.enable_tracing:
561                execution_metadata["span_error"] = True
562                tracer.log_event(
563                    name="tool.error",
564                    trace_id=trace_id,
565                    parent_observation_id=parent_observation_id,
566                    metadata={
567                        "tool_name": tool_name,
568                        "error": error_msg,
569                        "error_type": type(e).__name__,
570                    },
571                )
572
573            # Return ToolOutput
574            return create_tool_output(
575                tool_name=tool_name,
576                success=False,
577                error=error_msg,
578                execution_time_ms=execution_time_ms,
579                metadata=execution_metadata,
580            )
