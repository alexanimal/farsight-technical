Pinecone client for managing vector database connections and queries.

This module provides a Pinecone client that manages connections to the vector database and provides methods for querying documents with metadata filters.

 17class PineconeClient:
 18    """Pinecone client for vector database operations.
 19
 20    This client manages connections to Pinecone and provides methods for
 21    querying vector embeddings with metadata filters. It supports querying
 22    documents from specified indexes and listing available indexes.
 23
 24    Example:
 25        ```python
 26        client = PineconeClient()
 27        await client.initialize()
 28        results = await client.query(
 29            index_name="companies",
 30            query_vector=[0.1, 0.2, ...],
 31            top_k=10,
 32            metadata_filter={"sector": "AI"}
 33        )
 34        await client.close()
 35        ```
 36
 37        Or use as a context manager:
 38        ```python
 39        async with PineconeClient() as client:
 40            results = await client.query(
 41                index_name="companies",
 42                query_vector=embedding,
 43                top_k=5
 44            )
 45        ```
 46    """
 47
 48    def __init__(
 49        self,
 50        api_key: Optional[str] = None,
 51        default_index: Optional[str] = None,
 52    ):
 53        """Initialize the Pinecone client.
 54
 55        Args:
 56            api_key: Optional Pinecone API key. If not provided, uses
 57                settings.pinecone_api_key.
 58            default_index: Optional default index name. If not provided, uses
 59                settings.pinecone_index. This can be overridden in query calls.
 60        """
 61        self._api_key = api_key or settings.pinecone_api_key
 62        self._default_index = default_index or settings.pinecone_index
 63        self._client: Optional[Pinecone] = None
 64
 65    async def initialize(self) -> None:
 66        """Initialize the Pinecone client connection.
 67
 68        Raises:
 69            ValueError: If API key is not provided.
 70            Exception: If connection to Pinecone fails.
 71        """
 72        if self._client is not None:
 73            logger.warning("Pinecone client already initialized, skipping re-initialization")
 74            return
 75
 76        if not self._api_key:
 77            raise ValueError(
 78                "Pinecone API key is required. Set PINECONE_API_KEY in environment or pass api_key parameter."
 79            )
 80
 81        try:
 82            self._client = Pinecone(api_key=self._api_key)
 83            logger.info("Pinecone client initialized successfully")
 84        except Exception as e:
 85            logger.error(f"Failed to initialize Pinecone client: {e}")
 86            raise
 87
 88    async def close(self) -> None:
 89        """Close the Pinecone client connection."""
 90        if self._client is not None:
 91            # Pinecone client doesn't have an explicit close method in v8+
 92            # but we can clear the reference
 93            self._client = None
 94            logger.info("Pinecone client closed")
 95
 96    async def __aenter__(self) -> "PineconeClient":
 97        """Async context manager entry."""
 98        await self.initialize()
 99        return self
100
101    async def __aexit__(self, exc_type, exc_val, exc_tb) -> None:
102        """Async context manager exit."""
103        await self.close()

121    def list_indexes(self) -> list[str]:
122        """List all available index names in the Pinecone project.
123
124        Returns:
125            List of index names.
126
127        Raises:
128            RuntimeError: If client is not initialized.
129            Exception: If listing indexes fails.
130
131        Example:
132            ```python
133            indexes = client.list_indexes()
134            print(f"Available indexes: {indexes}")
135            ```
136        """
137        if self._client is None:
138            raise RuntimeError(
139                "Pinecone client not initialized. Call initialize() first or use as context manager."
140            )
141
142        try:
143            # In Pinecone v8, list_indexes() returns a list of index objects
144            indexes = self._client.list_indexes()
145            # Extract index names from the response
146            if hasattr(indexes, "names"):
147                index_names = indexes.names()
148            elif isinstance(indexes, list):
149                index_names = [idx.name if hasattr(idx, "name") else str(idx) for idx in indexes]
150            else:
151                # Fallback: try to iterate
152                index_names = [idx.name for idx in indexes] if hasattr(indexes, "__iter__") else []
153
154            logger.debug(f"Found {len(index_names)} indexes: {index_names}")
155            return index_names
156        except Exception as e:
157            logger.error(f"Failed to list indexes: {e}")
158            raise

160    async def query(
161        self,
162        query_vector: list[float],
163        index_name: Optional[str] = None,
164        top_k: int = 10,
165        metadata_filter: Optional[dict[str, Any]] = None,
166        include_metadata: bool = True,
167        include_values: bool = False,
168        namespace: Optional[str] = None,
169    ) -> Any:
170        """Query the Pinecone index for similar vectors.
171
172        This method queries the specified index with a vector and optional
173        metadata filters to find the most similar documents.
174
175        Args:
176            query_vector: The query vector (embedding) to search for.
177            index_name: Name of the index to query. If not provided, uses
178                the default index from settings or initialization.
179            top_k: Number of results to return (default: 10).
180            metadata_filter: Optional metadata filter dict. Supports Pinecone
181                filter expressions. Example: {"sector": "AI", "status": "active"}
182            include_metadata: Whether to include metadata in results (default: True).
183            include_values: Whether to include vector values in results (default: False).
184            namespace: Optional namespace to query within the index.
185
186        Returns:
187            Query response object containing matches with scores, metadata, and IDs.
188            The response has a 'matches' attribute containing the results.
189
190        Raises:
191            RuntimeError: If client is not initialized.
192            ValueError: If index_name is not provided and no default is set.
193            Exception: If query execution fails.
194
195        Example:
196            ```python
197            # Simple query
198            results = await client.query(
199                query_vector=[0.1, 0.2, 0.3, ...],
200                top_k=5
201            )
202
203            # Query with metadata filter
204            results = await client.query(
205                query_vector=embedding,
206                index_name="companies",
207                top_k=10,
208                metadata_filter={"sector": "AI", "founded_year": {"$gte": 2020}}
209            )
210
211            # Access results
212            for match in results.matches:
213                print(f"ID: {match.id}, Score: {match.score}")
214                if match.metadata:
215                    print(f"Metadata: {match.metadata}")
216            ```
217        """
218        if self._client is None:
219            raise RuntimeError(
220                "Pinecone client not initialized. Call initialize() first or use as context manager."
221            )
222
223        index_name = index_name or self._default_index
224        if not index_name:
225            raise ValueError(
226                "index_name must be provided either as parameter or set as default_index."
227            )
228
229        try:
230            index = self._client.Index(index_name)
231
232            # Build query parameters - Pinecone v8 uses different parameter names
233            query_params: dict[str, Any] = {
234                "vector": query_vector,
235                "top_k": top_k,
236                "include_metadata": include_metadata,
237                "include_values": include_values,
238            }
239
240            if metadata_filter:
241                query_params["filter"] = metadata_filter
242
243            if namespace:
244                query_params["namespace"] = namespace
245
246            # Note: Pinecone query operations are typically synchronous
247            # but we keep the async interface for consistency
248            response = index.query(**query_params)
249            logger.debug(
250                f"Query executed successfully on index '{index_name}': "
251                f"top_k={top_k}, filter={metadata_filter is not None}"
252            )
253            return response
254        except Exception as e:
255            logger.error(f"Query execution failed on index '{index_name}': {e}")
256            raise

322        """Get information about a specific index.

326
327        Returns:
328            Dictionary containing index information (dimension, metric, etc.).
329
330        Raises:

332            ValueError: If index_name is not provided and no default is set.
333            Exception: If getting index info fails.
334
335        Example:
336            ```python
337            info = client.get_index_info("companies")
338            print(f"Index dimension: {info['dimension']}")
339            ```
340        """
341        if self._client is None:
342            raise RuntimeError(
343                "Pinecone client not initialized. Call initialize() first or use as context manager."
344            )
345
346        index_name = index_name or self._default_index

348            raise ValueError(
349                "index_name must be provided either as parameter or set as default_index."

364                ),
365            }

368            raise
